# Use the `<Search />` component to connect to the Zotero API

This example shows how to use the `<Search />` component to query Zotero's API. It supports searching by title, tags, and other fields available in Zotero.

<div className="row">
  <div className="col-sm-4">
```jsx
     <Search
  limitTo="advanced"
  source={{ customApi: ZoteroService }}
  operators={{"_contains": "Contains"}}
  fieldList={{
    q: "Search Query",
    tag: "Tags",
  }}
 resultItemTemplate={(item) => {
    // Se la nuova logica di parse restituisce un oggetto con campi
    //   { id, name, altLabel, biblio, geometry, ... }
    // adattiamo di conseguenza
    if (!item) {
      return <div className="text-muted">No results found.</div>;
    }

    return (
      <div key={item.id} className="card my-3">
        <div className="card-body">
          {/* Nome del luogo o titolo (dipende dal tuo parse) */}
          <h5 className="card-title">{item.name}</h5>

          {item.altLabel && (
            <p>
              <strong>Alt Label:</strong> {item.altLabel}
            </p>
          )}

          {item.biblio && item.biblio.length > 0 ? (
            <>
              <p><strong>Bibliographic References:</strong></p>
              <ul>
                {item.biblio.map((bib, idx) => (
                  <li key={bib.key || idx}>
                    <strong>{bib.title}</strong> â€” {bib.authors}, {bib.date}
                  </li>
                ))}
              </ul>
            </>
          ) : (
            <p>No bibliography found.</p>
          )}
        </div>
      </div>
    );

}}
/>

```
