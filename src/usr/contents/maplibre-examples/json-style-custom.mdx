import { MapLibre, Documenter } from "../../../modules/scms.js"

### Building a map from a _custom_ JSON style

<Documenter>
  <MapLibre 
    center="37.00416,37.48551,9"
    mapStyle="/data/ksa_custom.json"
  />
</Documenter>


Content of `ksa_custom.json`:
```json
...
{
  "id": "ksa",
  "source": "vectorLyr",
  "source-layer": "scms_ksa",
  "type": "circle",
  "metadata": {
    "label": "Kahramanmaraş Survey Sites",
    "searchInFields": {
      "Item_Label": "Label",
      "Site_Name": "Site name",
      "Site_Description": "Site description"
    },
    "popupTemplate": "<div><h4>${Site_Name}</h4><p>${Site_Description}</p><small>${Item_Label}</small></div>"
  },
  "paint": {
    "circle-color": "#ff0000",
    "circle-opacity": 0.5,
    "circle-radius": 5,
    "circle-stroke-width": 1.5,
    "circle-stroke-color": "#000000"
  }
...
```
